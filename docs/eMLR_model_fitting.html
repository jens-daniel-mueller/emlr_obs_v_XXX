  select(data_source, basin, gamma_slab, model, era, AIC, BIC, rmse) %>% 
  arrange(era) %>% 
  group_by(data_source, basin, gamma_slab, model) %>% 
  mutate(eras = paste(lag(era), era, sep = " --> "),
         rmse_sum = rmse + lag(rmse),
         aic_sum = AIC + lag(AIC),
         bic_sum = BIC + lag(BIC)
         ) %>% 
  ungroup() %>% 
  select(-c(era)) %>% 
  drop_na()

# chose which criterion is applied
