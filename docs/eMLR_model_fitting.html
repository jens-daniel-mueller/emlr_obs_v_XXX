# calculate RMSE sum for adjacent eras
lm_all_eras <- GLODAP_glanced_clean  %>%
  select(basin, gamma_slab, model, era, AIC, BIC, rmse) %>% 
  arrange(era) %>% 
  group_by(basin, gamma_slab, model) %>%
